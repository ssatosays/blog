Bashで実行時間を計測: サンプルコード #15

August 06, 2023
Dev 
#!/bin/bash

BEGIN=$(date +'%s.%3N')
# work
sleep 10
END=$(date +'%s.%3N')

DURING=$(echo "scale=3; ${END} - ${BEGIN}" | bc)
echo "BEGIN: ${BEGIN}"
echo "END: ${END}"
echo "DURING: ${DURING}"
