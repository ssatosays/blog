FTPファイルアップロードの自動化: サンプルコード #17

August 06, 2023
Dev 
#!/bin/bash

# FTP server settings
SERVER="127.0.0.1"
PORT="21"
USER="username"
PW="password"
REMOTE_DIR="/public"

# Temporary file
TMP_FILE="tmp"
touch $TMP_FILE

# FTP command sequence
FTP_COMMANDS=$(cat << EOF
open $SERVER $PORT
user $USER $PW
passive
prompt
cd $REMOTE_DIR
mput $TMP_FILE
ls
bye
EOF
)

# Run
echo "$FTP_COMMANDS" | ftp -n
