#!/bin/bash

one_day_ago=$(date --date '1 day ago' "+%Y%m%d")
today=$(date "+%Y%m%d")
files=$(ls ~/ | grep -e .xls$ -e .xlsx$ | grep -v ${one_day_ago} | grep -v ${today})
if [ ! -z "${files}" ]; then
  rm -rf ${files}
fi
